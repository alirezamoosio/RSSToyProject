Initial Database Setup:
First of all in order to enable saving Persian strings do the following:
1. run this in terminal: sudo gedit /etc/mysql/my.cnf
2. at the end of the file copy the string below:
[client]
default-character-set=utf8

[mysql]
default-character-set=utf8


[mysqld]
collation-server = utf8_unicode_ci
init-connect='SET NAMES utf8'
character-set-server = utf8
****
Now run the codes below:
3. sudo /etc/init.d/mysql stop 
4. sudo /etc/init.d/mysql restart
****
log in to your mySQL and run the SQL codes below
CREATE DATABASE RSSDatabase CHARACTER SET utf8 COLLATE utf8_general_ci;
use RSSDatabase;
create table if not exists Site(RSSLink varchar(50), contentClass varchar(50),site varchar(50), dateFormat varchar(50), primary key(RSSLink));
create table if not exists News(title varchar(1000) not null, date timestamp, link varchar(300) not null , content text, site varchar(50), ID int, primary key (ID));
